# Get the HTTP Port for the service

  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ template "ibm-dba-extshare-prod.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} | grep proxy | awk '{print $1}' )

# Get the HTTPS Port for the service

  export NODE_PORT_HTTPS=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[1].nodePort}" services {{ template "ibm-dba-extshare-prod.fullname" .  }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} | grep proxy | awk '{print $1}' )


  echo http://$NODE_IP:$NODE_PORT

  echo https://$NODE_IP:$NODE_PORT_HTTPS
