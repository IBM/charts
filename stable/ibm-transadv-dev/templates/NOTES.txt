Transformation Advisor is delivered as an interconnected set of pods and kubernetes services.
Get the application URL by running these commands:
export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ template "name" . }}-{{ .Values.transadvui.service.name }})
echo http://$NODE_IP:$NODE_PORT
