1. Launch the application from IBM Cloud Private Console. Workloads --> Helm Releases
2.Get the application URL's by running these commands:

# Get the HTTPS Port for the service
# Replace this with actual deployment service name for <Service Name>

  export NODE_PORT_HTTPS=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[1].nodePort}" services {{ template "ibm-dba-multicloud-prod.fullname" . }})

# Get the proxy IP for the cluster
  export NODE_IP=$(kubectl get nodes | grep proxy | awk '{print $1}')

  echo https://$NODE_IP:$NODE_PORT_HTTPS
