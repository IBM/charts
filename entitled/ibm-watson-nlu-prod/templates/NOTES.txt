1. Get the application URL by running these commands:
  echo "Port forward to nlu-server service on port 443"
  kubectl port-forward svc/{{ .Release.Name }}-ibm-watson-nlu-nluserver 4443:443 --namespace {{ .Release.Namespace }}

  echo "Sample curl request to get Keywords"
  echo "curl -k -X POST   'https://localhost:4443/natural-language-understanding/api/v1/analyze?version=2019-10-16' -H 'Cache-Control: no-cache'   -H 'Content-Type: application/json'     -d '{
    "text":"Washington D.C. is the capital of United States of America.",
    "features": {
       "keywords":{}
    }
}'"

echo "Sample curl request to get Sentiment"
echo "curl -k -X POST   'https://localhost:4443/natural-language-understanding/api/v1/analyze?version=2019-10-16' -H 'Cache-Control: no-cache'   -H 'Content-Type: application/json'     -d '{
  "text":"I love my example text.",
  "features": {
     "sentiment":{}
  }
}'"

echo "Sample curl request to get Keywords with Sentiment"
echo "curl -k -X POST   'https://localhost:4443/natural-language-understanding/api/v1/analyze?version=2019-10-16' -H 'Cache-Control: no-cache'   -H 'Content-Type: application/json'     -d '{
  "text":"Washington D.C. is the capital of United States of America.",
  "features": {
     "keywords":{"sentiment":true}
  }
}'"

echo "Sample curl request to get Entities with Sentiment"
echo "curl -k -X POST 'https://localhost:4443/natural-language-understanding/api/v1/analyze?version=2019-10-16' \
-H 'Cache-Control: no-cache' \
-H 'Content-Type: application/json' \
-d '{
  "text":"Washington D.C. is the capital of United States of America.",
  "features": {
     "entities": { "sentiment": true }
  },
  "language":"en"
}'"
