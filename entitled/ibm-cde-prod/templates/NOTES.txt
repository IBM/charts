Get the Dashboard as a Service URL by running:
  export POD_NAME=$(kubectl get pods -n {{ .Release.Namespace }} -l "app=daas-sb-proxy,release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl -n {{ .Release.Namespace }} port-forward $POD_NAME 3001:3001

Or, using this one-liner:
  kubectl port-forward $(kubectl get pods -n {{ .Release.Namespace }} -l "app=daas-sb-proxy,release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}") 3001:3001

and then visit https://localhost:3001

To view the logs for daas-sb-proxy:

  kubectl logs $(kubectl get pods -n {{ .Release.Namespace }} -l "app=daas-sb-proxy,release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")

To view the logs for daas-server:

  kubectl logs $(kubectl get pods -n {{ .Release.Namespace }} -l "app=daas-server,release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
